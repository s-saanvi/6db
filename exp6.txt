create procedure merge_rollcall_data()
begin
declare done int default false;
declare n_id int;
declare n_name varchar(25);
declare n_birthdate date;
declare n_cursor cursor for
select student_id, student_name , birthdate
from n_rollcall;
declare continue handler for not found
set done = true;
open n_cursor;
cursor_loop : loop
fetch n_cursor into n_id,n_name,n_birthdate;
if done then
leave cursor_loop;
end if;
if not exists (select 1 from o_rollcall where student_id=n_id) then
insert into o_rollcall(student_id,student_name,birthdate) values (n_id,n_name,n_birthdate);
end if;
end loop;
close n_cursor;
end;
